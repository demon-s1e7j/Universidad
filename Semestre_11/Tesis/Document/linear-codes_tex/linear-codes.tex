\chapter{Linear Codes}

\begin{definition}\label{linear_code}
A code $c$ that encodes $k$ bits of information into an $n$-bit space is called \textit{linear} if every possible string to be encoded can be represented as a $k$-dimensional binary vector $v$ and it is encoded by the operation $gv$, where $g$ is an $n \times k$ matrix with entries in $\mathbb{Z}_2$, independent of $v$, called a \textit{generator matrix}. The vectors that form $g$ are typically chosen as a basis for the subspace of codewords. Thus, every operation is performed modulo two and every possible codeword is a linear combination of these basis vectors. \cite{https://doi.org/10.48550/arxiv.quant-ph/9705052}
\end{definition}

\begin{theorem}\label{dual_matrix_of_a_linear_code}
Given an $[n, k]$ linear binary code $c$ with binary generator matrix $g \in m_{n \times k}$, there exists a matrix $p \in m_{(n - k)\times n}$ such that for any vector $s$, $g s \in c$ if and only if $p s = 0$. More generally, a vector $v \in \mathbb{F}^n_2$ is a codeword if and only if $pv = 0$.
\end{theorem}

\begin{proof}\label{proof_dual_matrix_of_a_linear_code}
By definition, a linear code $c$ is a $k$-dimensional subspace of $\mathbb{F}^n_2$. Therefore, we can consider its orthogonal complement $c^\perp \subseteq \mathbb{F}^n_2$, defined as
\begin{equation}
c^\perp = \left\{w \in \mathbb{F}^n_2 \mid \forall v \in c, \langle w, v \rangle = 0\right\}.
\end{equation}
It is known that the orthogonal complement has dimension $\dim(c^\perp) = \dim(\mathbb{F}^n_2) - \dim(c) = n - k$, so $c^\perp$ has a basis $\{p_1, p_2, \ldots, p_{n - k} \}$. We can then construct the matrix
\begin{equation}
p = \begin{pmatrix}p_1 \\ p_2 \\ \vdots \\ p_{n - k} \end{pmatrix}.
\end{equation}

\begin{itemize}
\item ($\implies$) Let $v \in c$. By the definition of $c^\perp$, we know that for each row $p_i$, $p_i v = 0$, and therefore $pv = 0$.
\item ($\impliedby$) Let $v \in \mathbb{F}^n_2$ such that $p v = 0$. This means that for every row $p_i \in p$, $p_i \cdot v = 0$, implying that $v \in (c^\perp)^\perp$. By Lemma~\ref{complement_of_orthogonal_complement}, we conclude that $v \in c$.
\end{itemize}
\end{proof}

\begin{lemma}\label{complement_of_orthogonal_complement}
Given a subspace $c \subseteq \mathbb{F}^n_2$ and its orthogonal complement $c^\perp$,
\begin{equation}
(c^\perp)^\perp = c.
\end{equation}

By the definition of $c^\perp$, we know that $\forall v \in c$, $v \in (c^\perp)^\perp$. Moreover, by calculating dimensions, we observe that
\begin{equation}
\dim((c^\perp)^\perp) = \dim(\mathbb{F}^n_2) - \dim(c^\perp) = n - (n - \dim(c)) = \dim(c).
\end{equation}
Since $c \subseteq (c^\perp)^\perp$ and both have the same dimension, they are the same subspace.
\end{lemma}

\begin{example}\label{first_example_linear_code}
For example, the repetition code can be thought of as a linear code with generator matrix
\begin{equation}
g = \begin{bmatrix}1 \\ 1 \\ 1\end{bmatrix}.
\end{equation}
It then receives any single bit of information (represented in this case as a vector, as noted earlier), producing:
\begin{equation}
\begin{bmatrix}1 \\ 1 \\ 1\end{bmatrix} \begin{bmatrix} 0 \end{bmatrix} = \begin{bmatrix} 0 \\ 0 \\ 0 \end{bmatrix};\quad \begin{bmatrix}1 \\ 1 \\ 1\end{bmatrix} \begin{bmatrix} 1 \end{bmatrix} = \begin{bmatrix} 1 \\ 1 \\ 1 \end{bmatrix}.
\end{equation}

We can define the parity matrix as
\begin{equation}
p = \begin{bmatrix}1 & 1 & 0 \\ 1 & 0 & 1\end{bmatrix}.
\end{equation}

Errors can be checked as follows:

\begin{itemize}
\item If $p s = (0, 0)$, there was no error.
\item If $p s = (1, 0)$, the error occurred in the second bit.
\item If $p s = (0, 1)$, the error occurred in the third bit.
\item If $p s = (1, 1)$, the error occurred in the first bit.
\end{itemize}
\end{example}
