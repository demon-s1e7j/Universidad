  \section{}

  Partimos desde la definición
  $$P_{ij}^{(\alpha)} \coloneq \frac{n_\alpha}{\left|G\right|} \sum_{g \in G} D_{ij}^{(\alpha)}(g^{-1})r(g)$$

  Y por tanto

  \begin{align*}
    P_{ij}^{(\alpha)} P_{kl}^{(\beta)} &= \left(\frac{n_\alpha}{\left|G\right|} \sum_{g \in G} D_{ij}^{(\alpha)}(g^{-1})r(g)\right)\left(\frac{n_\beta}{\left|G\right|} \sum_{h \in G} D_{kl}^{(\beta)}(h^{-1})r(h)\right)\\
    &= \frac{n_\alpha n_\beta}{\left|G\right|^2} \sum_{g, h \in G} D_{ij}^{(\alpha)}(g^{-1}) D_{kl}^{(\beta)}(h^{-1})r(g)r(h)\\
    &= \frac{n_\alpha n_\beta}{\left|G\right|^2} \sum_{g, h \in G} D_{ij}^{(\alpha)}(g^{-1}) D_{kl}^{(\beta)}(h^{-1})r(gh)\\
  \end{align*}

  Ahora podemos tomar una nueva variable $x = gh \implies h = g^{-1} x$ y por tanto

  \begin{align*}
    P_{ij}^{(\alpha)} P_{kl}^{(\beta)} &= \frac{n_\alpha n_\beta}{\left|G\right|^2} \sum_{g, x \in G} D_{ij}^{(\alpha)}(g^{-1}) D_{kl}^{(\beta)}((g^{-1} x)^{-1})r(x)\\
    &= \frac{n_\alpha n_\beta}{\left|G\right|^2} \sum_{g, x \in G} D_{ij}^{(\alpha)}(g^{-1}) D_{kl}^{(\beta)}(x^{-1}g)r(x)\\
    D_{kl}^{(\beta)} (x^{-1}g) &= \sum_{m = 1}^{n_\beta} D_{km}^{(\beta)}(x^{-1})D_{ml}^{(\beta)}(g)\\
    P_{ij}^{(\alpha)} P_{kl}^{(\beta)} &= \frac{n_\alpha n_\beta}{\left|G\right|^2} \sum_{g, x \in G} D_{ij}^{(\alpha)}(g^{-1}) \sum_{m = 1}^{n_\beta} D_{km}^{(\beta)}(x^{-1})D_{ml}^{(\beta)}(g) r(x)\\
    &= \frac{n_\alpha n_\beta}{\left|G\right|^2} \sum_{x \in G} r(x) \sum_{m = 1}^{n_\beta} D_{km}^{(\beta)}(x^{-1}) \left(\sum_{g \in G} D_{ml}^{(\beta)}(g) D_{ij}^{(\alpha)}(g^{-1}) \right)\\
    \sum_{g \in G} D_{ml}^{(\beta)}(g) D_{ij}^{(\alpha)}(g^{-1})  &= \frac{\left|G\right|}{n_\alpha} \delta_{\alpha\beta} \delta_{il}\delta_{mj}\\
    P_{ij}^{(\alpha)} P_{kl}^{(\beta)}&= \frac{n_\alpha n_\beta}{\left|G\right|^2} \sum_{x \in G} r(x) \sum_{m = 1}^{n_\beta} D_{km}^{(\beta)}(x^{-1}) \left(\frac{\left|G\right|}{n_\alpha} \delta_{\alpha\beta} \delta_{il}\delta_{mj}\right)\\
  \end{align*}

  Dado que tenemos $\delta_{mj}$ entonces esa sumatoria se reduce a simplemente el caso $m = j$

  \begin{align*}
    P_{ij}^{(\alpha)} P_{kl}^{(\beta)}&= \delta_{\alpha\beta} \delta_{il}\frac{n_\beta}{\left|G\right|} \sum_{x \in G} D_{kj}^{(\beta)}(x^{-1}) r(x) \\
    &= \delta_{\alpha\beta} \delta_{il}P_{kj}^{(\beta)}
  \end{align*}

  Esto ya es el resultado pues sabemos que en caso de que $\alpha \neq \beta$ entonces todo se va a $0$. Por lo tanto
  $$P_{ij}^{(\alpha)} P_{kl}^{(\beta)} = \delta_{\alpha\beta} \delta_{il} P_{kj}^{(\alpha)}$$

  \subsection{$P^2 = P$}

  \begin{align*}
    \left(P^{(\alpha)}\right)^2 &= \sum_{k, j = 1}^{n_{(\alpha)}} P_{kk}^\alpha P_{jj}^{(\alpha)}\\
    P_{kk}^{(\alpha)} P_{jj}^{(\alpha)} &= \delta_{\alpha\alpha} \delta_{kj} P_{kj}^{(\alpha)}\\
    \left(P^{(\alpha)}\right)^2 &= \sum_{k, j = 1}^{n_{(\alpha)}} \delta_{kj} P_{kj}^{(\alpha)}\\
    \left(P^{(\alpha)}\right)^2 &= \sum_{k = 1}^{n_{(\alpha)}} P_{kk}^{(\alpha)}\\
    \left(P^{(\alpha)}\right)^2 &= P^\alpha
  \end{align*}

  \subsection{$P^* = P$}

  \begin{align*}
    \left(P^{(\alpha)}\right)^* &= \left(\frac{n_\alpha}{\left| G \right|} \sum_{g \in G} \chi^{(\alpha)}(g^{-1})r(g)\right)^*\\
    &= \frac{n_\alpha}{\left| G \right|} \sum_{g \in G} \chi^{(\alpha)}(g)r(g^{-1})\\
  \end{align*}

  Pero esto es equivalente pues si tomamos $g^{-1} = h$ podemos cambiar de variable y recorrer el mismo espacio. Es decir
  \begin{align*}
    \left(P^{(\alpha)}\right)^* &= \frac{n_\alpha}{\left| G \right|} \sum_{h \in G} \chi^{(\alpha)}(h^{-1})r(h)\\
    \left(P^{(\alpha)}\right)^* &= P^{(\alpha)}
  \end{align*}

  \subsection{Multiplicidad $m_\alpha$}

  \begin{align*}
    tr(P^{(\alpha)})  &= tr\left(\frac{n_\alpha}{\left| G \right|} \sum_{g \in G} \chi^{(\alpha)}(g^{-1})r(g)\right)\\
    &= \frac{n_\alpha}{\left| G \right|} \sum_{g \in G} \chi^{(\alpha)}(g^{-1})tr\left(r(g)\right)
  \end{align*}

  Sabemos que la traza de $r$ sera el caracter completo que a su vez sabemos corresponde a la suma de los caracteres irrep con sus multiplicidades. Por lo tanto,
  $$tr(r(g)) = \sum_{\beta \in G} m_\beta \chi^{(\beta)}(g)$$

  Por lo tanto
  \begin{align*}
    tr(P^{(\alpha)}) &= \frac{n_\alpha}{\left| G \right|} \sum_{g \in G} \chi^{(\alpha)}(g^{-1})\sum_{\beta \in G} m_\beta \chi^{(\beta)}(g)\\
    tr(P^{(\alpha)}) &= \frac{n_\alpha}{\left| G \right|} \sum_{\beta \in G} m_\beta \sum_{g \in G} \chi^{(\alpha)}(g^{-1})\chi^{(\beta)}(g)
  \end{align*}

  Dado que los caracteres son ortogonales entre si esta suma solamente tiene sentido cuando $\alpha = \beta$

  \begin{align*}
    tr(P^{(\alpha)}) &= \frac{n_\alpha}{\left| G \right|} m_\alpha \sum_{g \in G} \chi^{(\alpha)}(g^{-1})\chi^{(\alpha)}(g)\\
    &= \frac{n_\alpha}{\left| G \right|} m_\alpha \left|G\right|\\
    &= n_\alpha m_\alpha\\
    m_\alpha &= \frac{1}{n\alpha} tr(P^{(\alpha)})
  \end{align*}

  \subsection{caracteristicas $P_kk^{(\alpha)}$}

  Las primeras caracteristicas ya las mostramos en las correspondientes demostraciones (es decir, utilizamos ambas cosas en los puntos de arriba). Sin embargo falta mostrar que $tr(P_{kk}^{(\alpha)}) = m_\alpha$.

  \begin{align*}
    tr(P_{kk}^{(\alpha)}) &= tr\left(\frac{n_\alpha}{\left|G\right|} \sum_{g \in G} D_{kk}^{(\alpha)}(g^{-1}) r(g)\right)\\
    tr(r(g)) &= \sum_{\beta \in G} m_\beta \chi^{(\beta)}(g) = \sum_{\beta \in G} m_\beta \sum_{l = 1}^{n_\alpha} D_{ll}^{(\beta)}(g)\\
    tr(P_{kk}^{(\alpha)}) &= \frac{n_\alpha}{\left|G\right|} \sum_{g \in G} D_{kk}^{(\alpha)}(g^{-1}) \sum_{\beta \in G} m_\beta \sum_{l = 1}^{n_\alpha} D_{ll}^{(\beta)}(g)\\
    tr(P_{kk}^{(\alpha)}) &= \frac{n_\alpha}{\left|G\right|} \sum_{\beta \in G} m_\beta \sum_{g \in G}  \sum_{l = 1}^{n_\alpha} D_{kk}^{(\alpha)}(g^{-1}) D_{ll}^{(\beta)}(g)\\
    tr(P_{kk}^{(\alpha)}) &= \frac{n_\alpha}{\left|G\right|} m_\alpha \sum_{g \in G} D_{kk}^{(\alpha)}(g^{-1}) D_{kk}^{(\alpha)}(g)\\
    tr(P_{kk}^{(\alpha)}) &= \frac{n_\alpha}{\left|G\right|} m_\alpha \frac{\left|G\right|}{n_\alpha}\\
    tr(P_{kk}^{(\alpha)}) &= m_\alpha\\
  \end{align*}

  Para demostrar que $v_{kj}^{(\alpha)}$ son linealmente independientes basta con mostrar que

  \begin{align*}
    \sum_{j = 1}^{n_\alpha} c_jv_{kj}^{(\alpha)} = 0 \implies \forall j;\ c_j = 0
  \end{align*}

  Desarrollando esto
  \begin{align*}
    \sum_{j = 1}^{n_\alpha} c_jv_{kj}^{(\alpha)} = 0\\
    \sum_{j = 1}^{n_\alpha} c_jP_{1j}^{(\alpha)}\omega_k = 0\\
  \end{align*}

  Ahora, sin perdida de generalidad tomemos para un $m \in \left\{1, \ldots, n_\alpha \right\}$ un $P_{m1}^{(\alpha)}$ para aplicarlo lo que nos daria.
  \begin{align*}
    \sum_{j = 1}^{n_\alpha} c_jP_{m1}^{(\alpha)}P_{1j}^{(\alpha)}\omega_k = P_{m1}0\\
    \sum_{j = 1}^{n_\alpha} c_j\left(\delta_{\alpha\alpha}\delta{mj}P_{11}\right)\omega_k = 0\\
    c_m P_{11}\omega_k = 0\\
    P_{11}\omega_k \neq 0 \implies c_m = 0\\
  \end{align*}

  Por lo tanto $$\forall m;\ c_m = 0$$

  Para este caso, nos piden la evolución de un vector $v_{kn}^{(\alpha)}$ (con $n \in \left\{1, \ldots, n_\alpha\right\}$ y $k, \alpha$ fijos) respecto a un operador del grupo $g$ dada la irrep $r$. Es decir, nos estan preguntando 
  $$r(g)v_{kn}^{(\alpha)} = r(g) P_{1n}^{(\alpha)} \omega_k = ?$$

  Por lo tanto podemos desarrollar
  \begin{align*}
    r(g)v_{kn}^{(\alpha)} &= r(g) P_{1n}^{(\alpha)} \omega_k\\
    &= r(g) \left(\frac{n_\alpha}{\left|G\right|} \sum_{h \in G} D_{1j}^{(\alpha)} (h^{-1}) r(h)\right)\omega_k\\
    &= \frac{n_\alpha}{\left|G\right|} \sum_{h \in G} D_{1j}^{(\alpha)} (h^{-1}) r(gh)\omega_k\\
    &= \frac{n_\alpha}{\left|G\right|} \sum_{x \in G} D_{1j}^{(\alpha)} (x^{-1}g) r(x)\omega_k\\
    D_{1j}^{(\alpha)}(x^{-1} g) &= \sum_{m = 1}^{n_\alpha} D_{1m}^{(\alpha)}(x^{-1}) D_{mj}^{(\alpha)} (g)\omega_k\\
    r(g)v_{kn}^{(\alpha)} &= \frac{n_\alpha}{\left|G\right|} \sum_{x \in G} \left(\sum_{m = 1}^{n_\alpha} D_{1m}^{(\alpha)}(x^{-1}) D_{mj}^{(\alpha)} (g) \right) r(x)\omega_k\\
    &= \sum_{m = 1}^{n_\alpha} D_{mj}^{(\alpha)} (g) \frac{n_\alpha}{\left|G\right|} \sum_{x \in G}  D_{1m}^{(\alpha)}(x^{-1})  r(x)\omega_k\\
    &= \sum_{m = 1}^{n_\alpha} D_{mj}^{(\alpha)} (g) P_{1m}^{(\alpha)}\omega_k\\
    &= \sum_{m = 1}^{n_\alpha} D_{mj}^{(\alpha)} (g) v_{kn}^{(\alpha)}
  \end{align*}

  Dado que ya mostramos que $v_{kn}^{(\alpha)}$ es una base esto demuestra que para la acción de un grupo este evoluciona respecto a aplicar la matriz $D^{(\alpha)}(g)$

  $$r(g) = P\otimes D^{(2)}(g)$$

$$\chi^r(g) = \chi_{perm}(g) \cdot \chi_{vec}(g)$$

Podemos dividir por clases de conjugación basandonos en la geometria del triangulo con el que estamos trabajando (y con pistas de la tabla anterior). Sin analizamos en estas clases tenemos:

\begin{itemize}
  \item \textbf{Identidad ($e$):} Los 3 átomos quedan fijos ($\chi_{perm}=3$) y la traza del desplazamiento local es 2. Total: $\chi^r(e) = 3 \times 2 = 6$.
  \item \textbf{Rotaciones ($\rho, \tau$):} Ningún átomo queda fijo ($\chi_{perm}=0$). Por tanto, $\chi^r(rot) = 0 \times (-1) = 0$.
  \item \textbf{Reflexiones ($\sigma_i$):} Solo el átomo sobre el eje queda fijo ($\chi_{perm}=1$). La traza de una reflexión local en 2D es 0. Total: $\chi^r(refl) = 1 \times 0 = 0$.
\end{itemize}

Obtenemos así el vector de caracteres $\chi^r = (6, 0, 0)$. Para encontrar la descomposición en irreps $V^r = \bigoplus m_\alpha V^{(\alpha)}$, aplicamos la fórmula de reducción $m_\alpha = \frac{1}{|G|} \sum \chi^{(\alpha)}(g)^* \chi^r(g)$:
\begin{align*}
m_1 &= \frac{1}{6} [ 1(6)(1) + 2(0)(1) + 3(0)(1) ] = 1 \\
m_2 &= \frac{1}{6} [ 1(6)(2) + 2(0)(-1) + 3(0)(0) ] = 2 \\
m_3 &= \frac{1}{6} [ 1(6)(1) + 2(0)(1) + 3(0)(-1) ] = 1
\end{align*}
Por lo tanto, la descomposición es: $V^r = V^{(1)} \oplus 2V^{(2)} \oplus V^{(3)}$.

Para este caso podemos simplemente darnos cuenta que:
\begin{align*}
  EP = E\\
  \implies \left(EP\right)^* = E^*\\
  \implies P^*E^* = E^*\\
\end{align*}

y dado que esta es una proyección esto implica

$$\implies PE = E$$

\subsection{$(P - E)$ es una proyección}

\subsubsection{$(P - E)^* = (P - E)$}

En este caso simplemente se puede aprovechar las caracteristicas de $(^*)$ y que $P$ y $E$ son proyecciónes (es decir $P^* = P \land E^* = E$)

\begin{align*}
  (P - E)^* &= P^* - E^*\\
  (P - E)^* &= P - E\\
\end{align*}

\subsubsection{$(P - E)^2 = (P - E)$}

De modo similar podemos desarrollar
\begin{align*}
  (P - E)^2 &= (P - E)(P - E)\\
  &= P^2 - PE - EP + E^2\\
  P^2 &= P\\
  PE &= E\\
  EP &= E\\
  E^2 &= E\\
  &= P - E - E + E\\
  &= P - E
\end{align*}
